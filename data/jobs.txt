mission money
	name "Need Cash?"
	repeat
	job
	source
		attributes forge
	description "I need a credit, a credit is what i need...  here, have one billion"
	on accept
		payment 1000000000
		fail

mission "Boost Combat Rating"
	repeat
	job
	source
		attributes forge
	description "Boost your combat rating by 200,000 (enough to go from 0 to 12)."
	on accept
		"combat rating" += 200000
		fail
